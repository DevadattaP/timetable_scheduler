\section*{Introduction}
This formulation represents the scheduling problem for the IIM EPGP Quarter II, accommodating different sections, courses, and time slots and faculties. The objective is to create an optimal schedule that minimizes conflicts and maximizes resource utilization.

\section{Sets and Indices}
\begin{itemize}
    \item $s \in S$: Set of Sections.
    \item $c \in C$: Set of Courses.
    \item $t \in T_s$: Set of Time Slots available for section $s$.
    \item $f \in F$: Set of Faculty members.
    \item $d \in D$: Set of unique Calendar Dates in the quarter.
    \item $w \in W$: Set of ISO Week numbers in the quarter.
    \item $g \in G$: Set of course conflict groups.
    \item $C_g \subseteq C$: Set of courses in conflict group $g$.
    \item $S_g \subseteq S$: Set of sections to which conflict group $g$ applies.
\end{itemize}

\section{Parameters}
\begin{itemize}
    \item $R_{s,c}$: Required number of sessions for section $s$ in course $c$.
    \item $Faculty(s,c)$: The specific faculty member $f$ assigned to teach course $c$ to section
$s$.
    \item $Date(s, t)$: The calendar date $d$ associated with slot $t$ for section $s$.
    \item $Week(s, t)$: The week number $w$ associated with slot $t$ for section $s$.
    \item $Unavailable(f, d)$: A binary flag that is 1 if faculty $f$ is unavailable on date $d$, and 0
otherwise.
    \item $M_f$: Maximum number of sessions faculty $f$ can teach in a day.
\end{itemize}

\section{Decision Variables}
\begin{itemize}
    \item $x_{s,c,t} \in \{0, 1\}$: 1 if section $s$ is assigned course $c$ at time slot $t$; 0 otherwise.
    \item $y_{s,c,w} \in \{0, 1\}$: Auxiliary variable; 1 if section $s$ has at least one session of course
$c$ in week $w$; 0 otherwise.
    \item $p_{s,c,w} \in \{0, 1\}$: Penalty variable; 1 if course $c$ is taught to section $s$ in both
week $w$ and week $w+1$; 0 otherwise.
\end{itemize}

\section{Objective Function}
\begin{equation}
\text{Minimize} \quad Z = \sum_{s \in S} \sum_{c \in C} \sum_{w \in W} p_{s,c,w}
\end{equation}
The goal is to minimize the total number of "Alternate Weekend" rule violations across all sections
and courses.

(For Hard constraint problem, there is no separate objective function, just to find feasible solution
which satisfies all the constraints)

\section{Constraints}
\begin{itemize}
    \item \textbf{H1: Session Requirement}
    \begin{equation}
    \sum_{t \in T_s} x_{s,c,t} = R_{s,c} \quad \forall s \in S, c \in C
    \end{equation}
    Every section must complete the exact number of required sessions for
every assigned course.

    \item \textbf{H2: One Course per Slot (Zero Slack)}
    \begin{equation}
    \sum_{c \in C} x_{s,c,t} = 1 \quad \forall s \in S, t \in T_s
    \end{equation}
    Each available time slot for a section must be occupied by
exactly one course.
    
    \item \textbf{H3: No Faculty Cloning}
    \begin{equation}
    \sum_{(s,c) : Faculty(s,c)=f} x_{s,c,t} <= 1 \quad \forall f \in F, t \in T_s \forall s \in S
    \end{equation}
    A faculty member cannot teach more than one section at the same
physical time.

    \item \textbf{H4: Maximum Daily Workload}
    \begin{equation}
    \sum_{s \in S} \sum_{c \in C} \sum_{t \in T_s : Date(s,t)=d, Faculty(s,c)=f} x_{s,c,t} <= M_f \quad \forall f \in F, d \in D
    \end{equation}
    A faculty member can teach a maximum of $M_f$ sessions per
calendar date across all sections.

    \item \textbf{H5: Course Spacing (Daily)}
    \begin{equation}
    \sum_{t \in T_s : Date(s,t)=d} x_{s,c,t} <= 1 \quad \forall s \in S, c \in C, d \in D
    \end{equation}
    A section cannot be scheduled for the same course more than once
on the same day.

    \item \textbf{H6: Faculty Unavailability}
    \begin{equation}
    x_{s,c,t} = 0 \quad \text{if } Unavailable(Faculty(s,c), Date(s,t)) = 1 \quad \forall s \in S, c \in C, t \in T_s
    \end{equation}
    Classes cannot be scheduled if the assigned faculty is unavailable on
that date.

    \item \textbf{H7: Course Conflict Groups with Section Scope}
    \begin{equation}
    \sum_{s \in S_g} \sum_{c \in C_g} x_{s,c,t} \leq 1 
    \quad \forall g \in G,\; \forall t \in T
    \end{equation}
    For each conflict group $g$, at most one course from the group can be scheduled at any given time slot across the selected sections $S_g$.

    \item \textbf{Hard Alternate Weekend Constraint (Optional)}
    \begin{equation}
    y_{s,c,w} + y_{s,c,w+1} <= 1 \quad \forall s \in S, c \in C, w \in \{1, \dots, |W|-1\}
    \end{equation}
    This constraint strictly prohibits scheduling the same course for the same section in consecutive weeks. This is the constraint that typically causes INFEASIBILITY because supply == demand leaves no room to avoid consecutive weeks.
\end{itemize}

\section{Soft Constraints (Linearized Penalties)}
To minimize consecutive-week scheduling, we link the assignment variable $x$ to the week-indicator $y$, and subsequently to the penalty $p$.
\begin{itemize}
    \item \textbf{S1: Weekly Activity Identification (Lower Bound)}
    \begin{equation}
    y_{s,c,w} >= x_{s,c,t} \quad \forall s, c, w, t \in \{t \in T_s : Week(s,t) = w\}
    \end{equation}
    For each section, course, and week, the indicator variable $y_{s,c,w}$ is forced to 1 if at least one
session is scheduled during that week.

    \item \textbf{S2: Weekly Activity Identification (Upper Bound)}
    \begin{equation}
    y_{s,c,w} <= \sum_{t \in T_s : Week(s,t) = w} x_{s,c,t} \quad \forall s, c \in C, w \in W
    \end{equation}
    To ensure $y_{s,c,w}$ remains 0 if no sessions are scheduled in a given week, we bound it by the
sum of assignments within that week.

    \item \textbf{S3: Consecutive Week Penalty Logic}
    
    The penalty variable $p_{s,c,w}$ is triggered only when a course is taught in both week $w$ and
week $w+1$ for the same section.
    \begin{equation}
    p_{s,c,w} >= y_{s,c,w} + y_{s,c,w+1} - 1 \quad \forall s, c, w \in \{1, \dots, |W|-1\}
    \end{equation}
    If both $y_{s,c,w} = 1$ and $y_{s,c,w+1} = 1$, then $1 + 1 - 1 = 1$. This
forces $p_{s,c,w} >= 1$, triggering the penalty. If either week has no sessions (0), the right-hand side is $\leq 0$. Since $p$ is
a binary variable being minimized in the objective function, it will naturally fall to
0.

\end{itemize}
